Par exemple :

```python
import os

def read_file(path):
    """
    Read the content of a file and return it as a string.
    
    Parameters:
    path (str): The path to the file.
    
    Returns:
    str: The content of the file. If the file is not found, returns None.
    """
    if not os.path.exists(path):
        print(f"File {path} does not exist.")
        return None
    
    try:
        with open(path, 'r') as file:
            return file.read()
    except Exception as e:
        print(f"An error occurred: {e}")
        return None

# Usage
read_result = read_file("E:\\Mémoire Claude\\CLAUDE_RESURRECTION.md")
if read_result is not None:
    print(read_result)
```
